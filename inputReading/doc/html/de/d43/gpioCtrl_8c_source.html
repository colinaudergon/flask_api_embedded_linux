228 {
229  int ret;
230  int status;
231  struct gpiohandle_request req;
232  strcpy(chrdev_name, gpio_chip);
233  fd = open(chrdev_name, 0);
234  if (fd == -1)
235  {
236  ret = -errno;
237  fprintf(stderr, "Failed to open %s\n", chrdev_name);
238 
239  status = -1;
240  }
241  else
242  {
243  req.lineoffsets[0] = gpio_line;
244  req.flags = GPIOHANDLE_REQUEST_INPUT;
245  snprintf(req.consumer_label, sizeof(req.consumer_label), "btn_%u", gpio_line);
246  req.lines = 1;

248  // Request GPIO line
249  ret = ioctl(fd, GPIO_GET_LINEHANDLE_IOCTL, &req);
250  if (ret == -1)

252  ret = -errno;
253  fprintf(stderr, "Failed to issue GET LINEHANDLE IOCTL (%d)\n", ret);
254  status = -1;
255  }
256  // Close GPIO device (not needed after acquiring the line handle)

261  }
262  else
263  {
264  status = 0;
265  }
266  }
267  struct gpio_desc gpio;
268  strcpy(gpio.gpiobank, gpio_chip);
